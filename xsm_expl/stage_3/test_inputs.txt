begin
    read(a);
    read(b);
    read(c);
    if (a < b) then
        if (b < c) then 
            write(c);
        else
            write(b);
        endif;
    else
        if (a < c) then
            write(c);
        else
            write(a);
        endif;
    endif;
end;




begin
    read(a);
    read(b);
    while(a>0) do
        write(a);
        while(b>0) do
            write(b);
            b=b-1;
        endwhile;
        b=a;
        a=a-1;
    endwhile;
end;
